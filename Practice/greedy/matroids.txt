def is_independent(S, bags):
	# Constraint 1
    if len(S) > len(bags):
        return False
	
	# Constraint 2
    gems_sorted = sorted([m for m, v in S], reverse=True)
    bags_sorted = sorted(bags, reverse=True)
    return all(gems_sorted[i] <= bags_sorted[i] for i in range(len(S)))
